 pwd: to check in which folder you are in 
 ls: to list all the files inside that folder 
 ls -la: to list all the files inside that folder even the hidden files too 
 cd 'folder name': to move into that folder 
 cd .. : to move one level up in the folder 
 cd - : back to where you were earlier 
 mkdir 'folder name': to create that folder 
 touch 'filename': to create the file 
 echo 'text' > file : to write inside the file 
 echo -e 'text' > file: to write inside the file that have multiple lines 
 cat 'file name': to read that file 
 grep 'text' filename: to find that text inside the file 
 find . -name 'filename': to find the file 
 rm 'filename' : to delete the file 
 rm -r 'foldername': to delete the folder 
 cat > 'filename' << 'EOF' text EOF: to write multiple line (different option)
